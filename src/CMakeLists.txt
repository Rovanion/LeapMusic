add_executable (Sample Sample.cpp)

if(APPLE)
	target_link_libraries (Sample ${Leap_lib}/libLeap.dylib)
elseif(WIN32)
	if (${arch} STREQUAL "i386")
		target_link_libraries (Sample ${Leap_lib}/x86/libLeap.dll)
	else()
		target_link_libraries (Sample ${Leap_lib}/x64/libLeap.dll)
	endif()
else()
	if (${arch} STREQUAL "i386")
		target_link_libraries (Sample ${Leap_lib}/x86/libLeap.so)
	else()
		target_link_libraries (Sample ${Leap_lib}/x64/libLeap.so)
	endif()
endif()
